sg 2021-01-12 VUE

# 💌 vuepress-plugin-sakura

# 📎 项目配置也是很简单

npm install vuepress-plugin-sakura -D

or

cnpm install vuepress-plugin-sakura -D

// 只要把这个放进 config的plugins中就可以了 有木有很简单
 ["sakura", {
        num: 20,  // 默认数量
        show: true, //  是否显示
        zIndex: -1,   // 层级
        img: {
          replace: false,  // false 默认图 true 换图 需要填写httpUrl地址
          httpUrl: '...'     // 绝对路径
        }     
    }]
